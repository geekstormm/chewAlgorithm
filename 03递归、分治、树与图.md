```C++
//leetcode 78子集
//给你一个整数数组 nums ，数组中的元素互不相同 。返回该数组所有可能的子集（幂集）。
//解集不能包含重复的子集。你可以按任意顺序返回解集。

class Solution {
public:
    vector<vector<int>> subsets(vector<int> &nums) {
        n = nums.size();
        recur(nums, 0);
        return res;
    }

private:
    int n;
    vector<int> chosen;
    vector<vector<int>> res;

    void recur(vector<int> &nums, int i) {
      //terminator
        if (i == n) {
            res.push_back(chosen);
            return;
        }

      	//proces current level and drill down
        recur(nums, i + 1);
        chosen.push_back(nums[i]);
        recur(nums, i + 1);
      
				//recur status
        chosen.pop_back();
    }
};
```

```C++
        return res;

        recur(i + 1);
        chosen.push_back(i);
```

```C++
//leetcode 46 全排列
class Solution {
public:
  vector<vector<int>> permute(vector<int> &nums) {
    n = nums.size();
    used = vector<bool>(n, false);
    recur(nums, 0);
    return res;
  }

private:
  int n;
  vector<bool> used;
  vector<int> permu;
  vector<vector<int>> res;

  void recur(vector<int> &nums, int pos) {
    if (pos == n) {
      res.push_back(permu);
      return;
    }

    for (int i = 0; i < n; ++i) {
      if (!used[i]) {
        permu.push_back(nums[i]);
        used[i] = true;
        recur(nums, pos+1);
        used[i] = false;
        permu.pop_back();
      }
    }
  }
};
```

```C++
class Solution {
public:
    ListNode *mergeKLists(vector<ListNode *> &lists) {
        return partion(lists, 0, lists.size() - 1);
    }

private:
    ListNode *partion(vector<ListNode *> &Lists, int left, int right) {
        if (left == right) return Lists[left];

        if (left < right) {
            int mid = (left + right) >> 1;
            ListNode *list1 = partion(Lists, left, mid);
            ListNode *list2 = partion(Lists, mid + 1, right);

            return merge(list1, list2);
        } else {
            return nullptr;
        }
    }

    ListNode *merge(ListNode *l1, ListNode *l2) {
        if (l1 == nullptr) return l2;
        if (l2 == nullptr) return l1;

        if (l1->val < l2->val) {
            l1->next = merge(l1->next, l2);
            return l1;
        } else {
            l2->next = merge(l1, l2->next);
            return l2;
        }
    }
};
```

### 二叉树的遍历***

树的先序、中序、后序一般用**递归**来求，树的先序遍历又称之为树的**深度优先遍历**。

层次序一般借助**队列**来求，树的层次遍历又称为树的**广度优先遍历**。

```C++
//leetcode 94 二叉树的中序遍历
class Solution {
public:
  vector<int> inorderTraversal(TreeNode *root) {
    dfs(root);
    return res;
  }

  void dfs(TreeNode *root) {
    if(root == nullptr) return;

    dfs(root->left);
    res.push_back(root->val);
    dfs(root->right);
  }

private:
  vector<int> res;
};
```

```c++
//迭代写法
vector<int> preorder(Node *root) {
  vector<int> res;
  if(root == nullptr) return res;

  stack<Node *> s;
  s.push(root);

  while (!s.empty()) {
    Node *node = s.top();
    s.pop();
    res.push_back(node->val);

    //每一层最后的元素先入栈，才能保证输出是根左右 
    for (int i = node->children.size() - 1; i >= 0; --i) {
      s.push(node->children[i]);
    }
  }

  return res;
}
```

```C++
class Solution {
public:
  vector<vector<int>> zigzagLevelOrder(TreeNode* root) {
    queue<TreeNode *> q;
    q.push(root);
    vector<vector<int>> res;
    if(root == nullptr) return res;

    int flag = false;
    while(!q.empty()) {
      flag = !flag;
      vector<int> part_res;
      size_t n = q.size();

      while(n--) {
        TreeNode* node = q.front();
        q.pop();
        part_res.push_back(node->val);

        if(node->left != nullptr)
          q.push(node->left);
        if(node->right != nullptr)
          q.push(node->right);
      }

      if(flag) {
        res.push_back(part_res);
      } else {
        vector<int> temp;
        for(int i = part_res.size() - 1; i >= 0; --i) {
          temp.push_back(part_res[i]);
        }
        res.push_back(temp);
      }
    }
    return res;
  }
```

```C++
class Solution {
public:
  TreeNode *buildTree(vector<int> &preorder, vector<int> &inorder) {
    this->preorder = preorder;
    this->inorder = inorder;
    int curRoot = 0;
    for (int i = 0; i < inorder.size(); ++i)

    return build(curRoot, 0, preorder.size() - 1);

  TreeNode *build(int &curRoot, int left, int right) {
    ///terminate
    if (left > right) return nullptr;

    ///process current logic
    int inRoot = hash[preorder[curRoot]];
    curRoot++;

    ///drill down
    TreeNode *newNode = new TreeNode(inorder[inRoot]);
    newNode->left = build(curRoot, left, inRoot - 1);
    newNode->right = build(curRoot, inRoot + 1, right);

    return newNode;
  }

private:
  unordered_map<int, int> hash;
  vector<int> preorder;
  vector<int> inorder;
};
```

```C++
class Solution {
public:
  TreeNode *buildTree(vector<int> &inorder, vector<int> &postorder) {
    this->inorder = inorder;
    this->postorder = postorder;
    int cur_root = inorder.size() - 1;

    for (int i = 0; i < inorder.size(); ++i) {
      hash[inorder[i]] = i;
    }

    return build(cur_root, 0, inorder.size() - 1);
  }

  TreeNode *build(int &cur_root, int left, int right) {
    if (left > right) return nullptr;

    int in_root = hash[postorder[cur_root]];
    --cur_root;

    TreeNode *node = new TreeNode(inorder[in_root]);
    node->right = build(cur_root, in_root + 1, right);
    node->left = build(cur_root, left, in_root - 1);

    return node;
  }

private:
  vector<int> inorder;
  vector<int> postorder;
  unordered_map<int, int> hash;
};
```

```C++
class Codec {
public:
  // Encodes a tree to a single string.
  string serialize(TreeNode *root) {
    vector<string> seq;
    dfs(root, seq);

    string res;
    for (int i = 0; i < seq.size() - 1; ++i) res += seq[i] + ",";
    res += seq[seq.size() - 1];
    return res;
  }

  void dfs(TreeNode *root, vector<string> &seq) {
    if (root == nullptr) {
      seq.push_back("null");
      return;
    }

    seq.push_back(to_string(root->val));
    dfs(root->left, seq);
    dfs(root->right, seq);
  }

  // Decodes your encoded data to tree.
  TreeNode *deserialize(string data) {
    auto seq = split(data, ",");
    int cur = 0;

    return restore(seq, cur);
  }

  TreeNode *restore(vector<string>& seq, int& cur) {
    if (seq[cur] == "null") {
      ++cur;
      return nullptr;
    }

    TreeNode *root = new TreeNode(stoi(seq[cur]));
    ++cur;
    root->left = restore(seq, cur);
    root->right = restore(seq, cur);

    return root;
  }

  vector<string> split(const string &data, string delimiter) {
    string s = data;
    vector<string> seq;
    size_t pos = 0;
    string token;

    while ((pos = s.find(delimiter)) != string::npos) {
      token = s.substr(0, pos);
      seq.push_back(token);
      s.erase(0, pos + delimiter.length());
    }
    seq.push_back(s);
    return seq;
  }
};
```

```C++
//树可以看成是一个连通且 无环 的 无向 图。

// 给定往一棵 n 个节点 (节点值 1～n) 的树中添加一条边后的图。添加的边的两个顶点包含在 1 到 n 中间，且这条附加的边不属于树中已存在的边。图的信

//
// 请找出一条可以删去的边，删除后可使得剩余部分是一个有着 n 个节点的树。如果有多个答案，则返回数组 edges 中最后出现的边。 


class Solution {
public:
    vector<int> findRedundantConnection(vector<vector<int>> &edges) {
        n = 0;
        for (auto &edge: edges) n = max(n, max(edge[0], edge[1]));

        has_cycle = false;
        to = vector<vector<int>>(n + 1, vector<int>());
        //visited = vector(n + 1, false);
        for (auto &edge: edges) {
            int x = edge[0], y = edge[1];
            to[x].push_back(y);
            to[y].push_back(x);
            visited = vector(n + 1, false);
            dfs(x, 0);
            if(has_cycle)   return edge;
        }
        return {};
    }

private:
    int n;
    bool has_cycle;
    vector<vector<int>> to;
    vector<bool> visited;

    //图的dfs判断有没有环
    void dfs(int x, int fa) {
        visited[x] = true;
        //出边数组访问x能到的周围点的方法
        for (int y: to[x]) {
            if (y == fa) continue;
            if (!visited[y]) dfs(y, x);
            else has_cycle = true;
        }
    }
};
```

```C++
      to[pre_course].push_back(course);
      in_degree[course]++;
    }

    queue<int> q;
    for (int i = 0; i < in_degree.size(); ++i)
      if (in_degree[i] == 0) q.push(i);

    vector<int> lessons;
    while (!q.empty()) {
      int pre = q.front();
      q.pop();
      lessons.push_back(pre);

      for (auto course: to[pre]) {
        in_degree[course]--;
        if (in_degree[course] == 0)
          q.push(course);
      }
    }

    if (lessons.size() == numCourses)
      return lessons;
    return {};
  }

private:
  vector<vector<int>> to;
  vector<int> inDeg;
  vector<int> in_degree;
};
```
