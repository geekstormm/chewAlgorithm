#### [25. K 个一组翻转链表](https://leetcode.cn/problems/reverse-nodes-in-k-group/)

```C++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* reverseKGroup(ListNode* head, int k) {
        ListNode* protect = new ListNode(0);
        ListNode* last = protect;

        while(head != nullptr) {
            //1.分组
            ListNode* end = getEnd(head, k);
            if(end == nullptr) break;
            ListNode *nextGroupHead = end->next;

            //2.每一组内部反转
            reverseList(head, nextGroupHead);

            //3.更新组与组之间的边
            last->next = end;
            head->next = nextGroupHead;
            last = head;
            head = nextGroupHead;
        }
        return protect->next;
    }

    //得到一组的end
    ListNode* getEnd(ListNode* head, int k) {
        while(head != nullptr) {
            k--;
            if(k == 0) return head;
            head = head->next;
        }
        return nullptr;
    }

    //组内反转链表
    void reverseList(ListNode* head, ListNode* stop) {
        ListNode* last = head;
        head = head->next;
        while(head != stop) {
            ListNode* next = head->next;
            head->next = last;
            last = head;
            head = next;
        }
    }
};
```

```C++
        while (r < s.size()) {
            //向右扩大窗口

                window[c]++;
                if(window[c] == need[c]) ++cnt;

                }
            }
```

```C++
class Solution {
public:
  vector<int> dailyTemperatures(vector<int>& temperatures) {
    int n = temperatures.size();
    vector<int> res(n, 0);
    stack<int> s;

    for(int i = 0; i < n; ++i) {
      while(!s.empty() && temperatures[i] > temperatures[s.top()]) {
        int out = s.top();
        res[out] = i - out;
        s.pop();
      }
      s.push(i);
    }

    return res;
  }
};
```

#### [402. 移掉 K 位数字](https://leetcode.cn/problems/remove-k-digits/)

从左到右便利，维护一个单调递增的栈即可。这里有两个细节，

- 最后栈里面的元素长度可能超过 n - k个，那么只取n - k个。
- 可能存在前导0，通过erase()和find_first_not_of()来处理。

看了别人的题解发现，vector和deque都比stack更好用hhh

```C++
class Solution {
public:
    string removeKdigits(string num, int k) {
        vector<char> st;
        for(char c : num) {
            while(st.size() && st.back() > c && k) {
                st.pop_back();
                k--;
            }
            st.push_back(c);
        }
        string res = string(st.begin(), st.end() - k);
        res.erase(0, res.find_first_not_of('0'));
        return res.empty() ? "0" : res;
    }
};
```

#### [255. 验证前序遍历序列二叉搜索树](https://leetcode.cn/problems/verify-preorder-sequence-in-binary-search-tree/)

二叉搜索树的前序遍历的特点在于：

-   如果出现递减序列，就是左子树
-   如果出现递增序列，就是右子树 

对先序遍历序列进行遍历，同时维护一个单调递减的栈，并通过单调栈来记录当前节点的最小值，如果后续的节点大于这个节点的值，则表示不合法。

```C++
bool verifyPreorder(vector<int>& preorder) {
  int n = preorder.size();
  stack<int> st;
  int preMin = INT_MIN;

  for(int i = 0; i < n; ++i) {
    if(preorder[i] < preMin) return false;
    while(!st.empty() && preorder[i] > st.top()) {
      preMin = st.top();
      st.pop();
    }
    st.push(preorder[i]);
  }
  return true;
}
```
